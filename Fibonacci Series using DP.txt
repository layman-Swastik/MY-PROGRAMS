#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int findByDP(int n,vector<int>& v)
{
    if(n == 0 || n == 1)
    {
        v[n] = n;
        return n;
    }
    if(v[n] == -1)
    {
        v[n] = findByDP(n-1,v)+findByDP(n-2,v);
        return v[n];
    }
    else
        return v[n];
}

int main() 
{
    int n;
    cin>>n;
    vector<int> v(n+1,-1);
    cout<<findByDP(n,v);
}
